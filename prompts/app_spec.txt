<project_specification>
  <project_name>vaultbank-banking-clone</project_name>

  <overview>
    A realistic VaultBank Bank clone for AI/automation task environments. All banking data is fake and state-defined via cookie-scoped backend.
    The application simulates a complete online banking experience with accounts, credit cards, loans, investments, bill pay, and transfers.
    No real financial connections - purely for experimental/testing purposes with both AI agents and human users.
  </overview>

  <technology_stack>
    <frontend>
      <framework>React 18</framework>
      <routing>React Router v6</routing>
      <styling>Chakra UI with custom VaultBank theme, Tailwind CSS for utilities</styling>
      <build_tool>Vite</build_tool>
      <language>JavaScript (JSX)</language>
    </frontend>
    <backend>
      <runtime>Python 3.11+ with FastAPI</runtime>
      <database>Cookie-scoped in-memory state (no database)</database>
      <architecture>State-based architecture with /api/state endpoints</architecture>
    </backend>
    <communication>
      <api>REST API with existing state endpoints + convenience endpoints</api>
      <credentials>Cookie-based user identification (user_id)</credentials>
      <mcp>Streamable HTTP MCP with get_accounts and execute_transfer tools</mcp>
    </communication>
  </technology_stack>

  <prerequisites>
    <environment_setup>
      - Node.js 18+ and npm for frontend
      - Python 3.11+ with virtual environment for backend
      - Backend already set up with FastAPI and state management
      - CORS configured for frontend origin
      - Existing /state-manage route must be preserved
    </environment_setup>
  </prerequisites>

  <database_schema>
    <tables>
      <data_state_structure>
        The application uses cookie-scoped state with no traditional database.
        All data is stored in a single JSON object per user with the following structure:

        data:
          accounts: Array[Account]
            - id: string (unique account identifier)
            - type: enum (checking, savings, cd, money_market)
            - nickname: string
            - account_number: string (last 4 digits)
            - routing_number: string
            - current_balance: number
            - available_balance: number
            - currency: string (default "USD")
            - opened_date: ISO date string
            - transactions: Array[Transaction]
              * id: string
              * date: ISO date string
              * description: string
              * amount: number
              * running_balance: number
              * category: string
              * type: enum (debit, credit, pending)
              * merchant?: string
            - statements: Array[Statement]
              * period: string (e.g., "January 2024")
              * url: string (PDF URL)
              * generated_date: ISO date string
            - status: enum (active, frozen, closed)
            - card_frozen: boolean

          credit_cards: Array[CreditCard]
            - id: string
            - card_number: string (last 4 digits)
            - cardholder_name: string
            - expiration_date: string (MM/YY)
            - current_balance: number
            - available_credit: number
            - credit_limit: number
            - payment_due_date: ISO date string
            - minimum_payment: number
            - statements: Array[Statement]
            - transactions: Array[Transaction] (similar to accounts)
            - status: enum (active, frozen, closed)
            - card_frozen: boolean

          loans: Array[Loan]
            - id: string
            - type: enum (auto, mortgage, personal)
            - loan_number: string
            - original_amount: number
            - current_balance: number
            - interest_rate: number (percentage)
            - monthly_payment: number
            - due_date: ISO date string
            - start_date: ISO date string
            - maturity_date: ISO date string
            - payment_schedule: Array[Payment]
              * date: ISO date string
              * amount: number
              * principal: number
              * interest: number
              * remaining_balance: number
              * status: enum (scheduled, paid, pending)
            - payment_history: Array[Payment]
            - status: enum (active, paid_off, in_application)

          investments: Investment
            - brokerage: BrokerageAccount
              * account_number: string
              * cash_balance: number
              * total_value: number
              * holdings: Array[Holding]
                - symbol: string
                - name: string
                - quantity: number
                - average_cost: number
                - current_price: number
                - current_value: number
                - gain_loss: number
                - gain_loss_percent: number
              - trades: Array[Trade]
                * id: string
                * symbol: string
                * type: enum (buy, sell)
                * quantity: number
                * price: number
                * total: number
                * date: ISO date string
            - retirement: Array[RetirementAccount]
              * Similar structure to brokerage
              * account_type: enum (401k, IRA, Roth_IRA)

          bill_pay: BillPay
            - payees: Array[Payee]
              * id: string
              * name: string
              * account_number: string
              * address: string
              * nickname?: string
            - scheduled_payments: Array[ScheduledPayment]
              * id: string
              * payee_id: string
              * amount: number
              * frequency: enum (once, weekly, monthly, yearly)
              * next_date: ISO date string
              * from_account_id: string
              * status: enum (active, cancelled, completed)
            - payment_history: Array[Payment]

          transfers: Transfers
            - history: Array[Transfer]
              * id: string
              * from_account_id: string
              * to_account_id: string (or external account info)
              * amount: number
              * date: ISO date string
              * status: enum (pending, completed, failed)
              * type: enum (internal, external)
            - scheduled: Array[ScheduledTransfer]
              * Similar to scheduled bill payments

          user_profile: UserProfile
            - first_name: string
            - last_name: string
            - email: string
            - phone: string
            - address: Address
              * street: string
              * city: string
              * state: string
              * zip: string
            - communication_preferences:
              * paperless: boolean
              * email_notifications: boolean
              * sms_notifications: boolean
            - security:
              * recent_logins: Array[LoginEvent]
                - timestamp: ISO date string
                - ip_address: string
                - device: string
                - location: string

        meta:
          - created_at: ISO date string
          - updated_at: ISO date string
          - version: number
          - type: string

        note: string or null
      </data_state_structure>
    </tables>
  </database_schema>

  <api_endpoints_summary>
    <existing_state_endpoints>
      - GET /api/state - Fetch current user state
      - PUT /api/state - Replace entire state
      - PATCH /api/state - Merge into existing state
      - DELETE /api/state - Reset state
      - GET /api/info - System and request info
    </existing_state_endpoints>

    <convenience_endpoints>
      - POST /api/transfer - Execute transfer
      - POST /api/payment - Process payment (bill or credit card)
      - GET /api/accounts - Get account summary
    </convenience_endpoints>

    <file_endpoints>
      - POST /api/files - Upload files
      - GET /api/files - List user files
      - GET /api/files/{filename} - Fetch file
    </file_endpoints>

    <mcp_tools>
      - get_accounts - Read accounts and balances
      - execute_transfer - Perform transfer with direct state modification
    </mcp_tools>
  </api_endpoints_summary>
